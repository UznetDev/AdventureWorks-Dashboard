With
		Member [Measures].[Fixed Moving Average] As
		'Avg (
		  { ParallelPeriod (<time_hierarchy_level, mdx_name,>, <time_periods, constant,>,
							<time_hierarchy, mdx_name,>.CurrentMember
	    ) : <time_hierarchy, mdx_name,>.CurrentMember }, [Measures].<measure_name, mdx_name,> )'

Select	{[Measures].<measure_name, mdx_name,>, [Measures].[Fixed Moving Average]} on Columns,
		<column_axis, mdx_set,> on Rows

From	<from_clause, mdx_name,>
Where	<where_clause, mdx_set,>
