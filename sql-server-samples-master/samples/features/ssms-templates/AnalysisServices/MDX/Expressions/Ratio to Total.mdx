-- =============================================
-- Ratio to Total
-- =============================================

With
		Member [Measures].[RatioToTotal] AS
		'[Measures].<measure_name, mdx_name,> /
		( [Measures].<measure_name, mdx_name,>, <hierarchy_name, mdx_name,>.<all_member, mdx_name,> )',
		Format_String = 'Percent'

Select	{[Measures].<measure_name, mdx_name,>, [Measures].[RatioToTotal]} on Columns,
		<column_axis, mdx_set,> on Rows

From	<from_clause, mdx_name,>
Where	<where_clause, mdx_set,>
