-- =============================================
-- Ratio to Parent
-- =============================================

With
		Member [Measures].[RatioToParent] AS
		'[Measures].<measure_name, mdx_name,> /
		( [Measures].<measure_name, mdx_name,>, <hierarchy_name, mdx_name,>.CurrentMember.Parent )',
		Format_String = 'Percent'

Select	{[Measures].<measure_name, mdx_name,>, [Measures].[RatioToParent]} on Columns,
		<column_axis, mdx_set,> on Rows

From	<from_clause, mdx_name,>
Where	<where_clause, mdx_set,>
